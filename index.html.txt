  <script>
    // small interactivity
    document.getElementById('year').textContent = new Date().getFullYear();

    const form = document.getElementById('bookingForm');
    const msg = document.getElementById('formMsg');
    form.addEventListener('submit', (e) => {
      e.preventDefault();
      const data = {
        name: document.getElementById('name').value.trim(),
        phone: document.getElementById('phone').value.trim(),
        address: document.getElementById('address').value.trim(),
        date: document.getElementById('date').value,
        service: document.getElementById('service').value,
        kg: document.getElementById('kg').value
      };

      if (!data.name || !data.phone || !data.address || !data.date) {
        msg.textContent = 'Please fill all required fields.';
        msg.classList.remove('text-green-600');
        msg.classList.add('text-red-600');
        return;
      }

      // simulate submission: save in localStorage
      const bookings = JSON.parse(localStorage.getItem('sw_bookings') || '[]');
      bookings.push(Object.assign({ id: Date.now() }, data));
      localStorage.setItem('sw_bookings', JSON.stringify(bookings));

      msg.textContent = 'Thanks! Pickup scheduled. We will call to confirm.';
      msg.classList.remove('text-red-600');
      msg.classList.add('text-green-600');
      form.reset();
      document.getElementById('kg').value = 5;
    });

    document.getElementById('clearBtn').addEventListener('click', () => {
      form.reset();
      document.getElementById('kg').value = 5;
      msg.textContent = '';
    });

    // mobile menu quick toggle
    const menuBtn = document.getElementById('menuBtn');
    menuBtn.addEventListener('click', () => {
      const nav = document.querySelector('header nav');
      nav.classList.toggle('hidden');
    });
  </script>
